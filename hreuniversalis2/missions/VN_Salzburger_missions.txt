vn_slz_missions_slot_1_a = {
    slot = 6
    generic = no
    ai = yes
    potential = {
        tag = SLZ
        NOT = { map_setup = map_setup_random }
    }
    has_country_shield = yes

    vn_slz_papal_backing = {
        icon = mission_religious
        position = 1
        required_missions = {  }
        trigger = {
			if = {
				limit = { has_dlc = "Emperor" }
				custom_trigger_tooltip = {
					tooltip = vn_slz_papal_backing.tt
					has_global_flag = investiture_controversy_active
				}
			}
			if = {
				limit = { 
					exists = PAP
					uses_papacy = yes
				}
				reverse_has_opinion = {
					who = PAP
					value = 100
				}
			}
			else_if = {
				limit = {
					NOT = { exists = PAP }
					uses_papacy = yes
				}
				papal_influence = 50
			}
        }
        effect = {
            if = {
				limit = { uses_papacy = yes }
				add_dip_power = 50
				add_religious_influence_effect_small = yes
				add_country_modifier = {
					name = "slz_papal_backing_mission"
					duration = 9125
				}
			}
			else = {
				custom_tooltip = vn_ase_elect_new_king.tt4	#Bypass
			}
        }
    }
    vn_slz_prince_archbishopric = {
        icon = mission_european_church
        position = 3
        required_missions = { vn_slz_papal_backing }
        trigger = {
			is_elector = yes
			government_rank = 5
			grown_by_development = 100
			government = theocracy
			devotion = 100
        }
        effect = {
			add_prestige = 20
			add_country_modifier = {
				name = "slz_prince_archbishopric_mission"
				duration = 9125
			}
        }
	}
}
