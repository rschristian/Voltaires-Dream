			if = {
				limit = {

			}
        }
        effect = {
            add_religious_influence_effect_large = yes

				limit = {
					has_estate = estate_church
				}

					estate = estate_church
				}
			}
			add_country_modifier = {
				name = "fra_alliance_with_church_mission"
				duration = 9125
			}
        }
	}
	vn_fra_sanction_crusades = {
		icon = mission_crusade
        required_missions = { vn_fra_alliance_with_church }

			is_in_war = {

			}
        }
        effect = {
			add_prestige = 10
            add_country_modifier = {
				name = "fra_sanction_crusades_mission"
				duration = 1825
			}
        }
	}
    vn_fra_deal_with_templars = {
        icon = mission_teutonic_knights
        required_missions = { vn_fra_sanction_crusades }
        trigger = {
			has_estate = estate_holy_orders

					}
				}
				NOT = {
					estate_territory = {
						estate = estate_holy_orders

					}
				}
			}
        }
        effect = {
