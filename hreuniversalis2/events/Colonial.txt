province_event = {
	id =  colonial.2
	title = "EVTNAME1420"
	desc = "EVTDESC1420"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			current_age = age_of_revolutions
			current_age = age_of_ideology
		}
		is_overseas = yes
		is_colony = no
		religion_group = christian
		owner = {
			religion_group = christian
			NOT = { has_country_modifier = narrowminded_modifier }
			NOT = { has_country_modifier = serfdom_modifier }
		}
		NOT = { has_province_modifier = founding_father }
		NOT = { has_province_modifier = loyalist_stronghold }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA1420"
		ai_chance = {
			factor = 100
		}
		add_province_modifier = {
			name = "founding_father"
			duration = -1
		}
	}
	
	option	= {
		name = "EVTOPTB1420"
		owner = {
			add_country_modifier = {
				name = narrowminded_modifier
				duration = 3650
			}
			add_country_modifier = {
				name = serfdom_modifier
				duration = 3650
			}
		}
		owner = { set_country_flag = treat_colonials_harshly }
	}
}

# Fur Trade Dwindles
province_event = {
	id = colonial.13
	title = "colonial.EVTNAME13"
	desc = "colonial.EVTDESC13"
	picture = MERCHANTS_TALKING_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		trade_goods = fur
		OR = {
			NOT = { has_province_flag = received_trade_goods }
			had_province_flag = { flag = received_trade_goods days = 3650 }
		}
		NOT = { has_empty_adjacent_province = yes }
		NOT = {
			any_neighbor_province = {
				owner = {
					has_reform = steppe_horde
				}
				controller = {
					has_reform = steppe_horde
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "colonial.EVTOPTA13"
		change_trade_goods = naval_supplies
	}
	option = {
		name = "colonial.EVTOPTC13"
		change_trade_goods = grain
	}
}

# Our colony has been struck by Roman Fever!
province_event = {
	id = colonial.15
	title = "colonial.EVTNAME15"
	desc = "colonial.EVTDESC15"
	picture = PLAGUE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_colony = yes
		continent = africa
		NOT = {
			owner = {
				has_country_modifier = roman_fever_timer
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			owner = {
				add_country_modifier = {
					name = roman_fever_timer
					duration = 5475
					hidden = yes
				}
			}
		}
	}
	
	option = {
		name = "colonial.EVTOPTA15"
		owner = { add_adm_power = -25 }
		add_province_modifier = {
			name = organised_colony
			duration = 1825
		}
	}
	option = {
		name = "colonial.EVTOPTB15"
		add_province_modifier = {
			name = quarantine_colony
			duration = 1825
		}
	}
	option = {
		name = "colonial.EVTOPTC15"
		add_province_modifier = {
			name = roman_fever_colony
			duration = 1825
		}
		random_neighbor_province = {
			limit = {
				is_colony = yes
				owner = {
					tag = ROOT
				}
			}
			add_province_modifier = {
				name = roman_fever_colony
				duration = 1825
			}
		}
	}
}

# Colonial Migration

	id = colonial.17
	title = "EVTNAME6401"
	desc = "EVTDESC6401"
	picture = COLONIZATION_eventPicture

	is_triggered_only = yes

	trigger = {
		colonysize = 300
		NOT = { colonysize = 500 }
		owner = {
			any_owned_province = {
				is_colony = yes
				colonysize = 500
				NOT = { province_id = ROOT }
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA6401"

			random_owned_province = {
				limit = {
					is_colony = yes
					colonysize = 500
					NOT = { province_id = ROOT }
				}
				add_colonysize = 200
			}
		}
	}
}
