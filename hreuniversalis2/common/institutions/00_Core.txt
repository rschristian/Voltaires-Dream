	penalty = 0.5
	bonus = {
		free_leader_pool = 1
	}

		OR = {
			AND = {
				owner = {
					NOT = {
						has_reform = tribal_despotism

						has_reform = steppe_horde
						government = native
					}
				}
				is_year = 1000
			}
			is_year = 1150
		}
	}
	
	# no need for events, as its there from the start.

			factor = 0.5
			scale = yes

		}

			factor = 0.1

				owner = { has_estate = estate_nobles }
			}
