non_catholic_rome = {
	potential = {
		normal_or_historical_nations = yes
	}
	
	#viewer = {
	#}
	
	trigger = {
		owner = {
			NOT = { religion = catholic }
			NOT = { religion = catholique }
			OR = {
				religion_group = christian
				religion_group = muslim
			}
		}
	}

	prestige = 0.5
	missionaries = 1
	
	#Province scope
	on_activation = {
	
	}
	
	on_deactivation = {
	
	}
	
}

religious_center = {
	potential = {
	}
	
	#viewer = {
	#}
	
	trigger = {
		if = {
			limit = {
				province_id = 2708 # Roma
			}
			religion = catholic
		}
		else_if = {
			limit = {
				province_id = 202 # Avignon
			}
			religion = catholique
		}
		else_if = {
			limit = {
				province_id = 781 # Praha
			}
			religion = hussite
		}
		else_if = {
			limit = {
				province_id = 59 # Wittenberg
			}
			religion = protestant
		}
		else_if = {
			limit = {
				province_id = 5012 # Medina
			}
			religion = sunni
		}
	}

	local_missionary_strength = -0.05
	
	#Province scope
	on_activation = {
	
	}
	
	on_deactivation = {
	
	}
	
}
