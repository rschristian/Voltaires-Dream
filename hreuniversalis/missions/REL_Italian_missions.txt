vn_italian_minor_group_2 = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		culture_group = latin
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = no
	
	itm_princely_rule = {
		position = 5
		icon = mission_italian_condottiere
		required_missions = { }
		trigger = {
			is_subject = no
			if = {
				limit = {
					OR = {
						government = monarchy
						has_reform = italian_monarchy
						has_government_attribute = has_dutch_election
						has_reform = signoria_reform
					}
				}
				custom_trigger_tooltip = {
					tooltip = itm_princely_rule_marriage
					calc_true_if = {
						all_ally = {
							marriage_with = ROOT
						}
						amount = 2
					}
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = itm_princely_rule_relations
					calc_true_if = {
						all_ally = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
						amount = 2
					}
				}
			}
		}
		effect = {
			add_legitimacy = 10
			add_devotion = 10
			add_republican_tradition = 10
			if = {
				limit = {
					NOT = { stability = 3 }
				}
				add_stability = 1
			}
			else = {
				add_adm_power = 100
			}
		}
	}
	itm_spheres_of_influence = {
		icon = mission_diplomatic_relation
		required_missions = { itm_princely_rule }
		trigger = {
			custom_trigger_tooltip = {
				tooltip = itm_two_guaranteed_countries
				calc_true_if = {
					all_country = {
						guaranteed_by = ROOT
					}
					amount = 2
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = itm_influential_government
				duration = 7300
			}
		}
	}
	itm_prelate_power = {
		icon = mission_bridge_the_heresy
		required_missions = { itm_spheres_of_influence }
		trigger = {
			if = {
				limit = {
					OR = {
						religion = catholic
						religion = catholique
					}
				}
				any_owned_province = {
					has_cardinal = yes
				}
			}
			else_if = {
				limit = {
					OR = {
						religion = protestant
						religion = anabaptist
						religion = hussite
					}
					has_dlc = "Common Sense"
				}
				num_of_aspects = 1
			}
			else_if = {
				limit = {
					religion = reformed
					has_dlc = "Wealth of Nations"
				}
				has_active_fervor = yes
			}
			else_if = {
				limit = {
					OR = {
						religion = orthodox
						religion = paulician
						religion = iconoclast
					}
				}
				patriarch_authority = 0.75
			}
			else_if = {
				limit = {
					religion = bogomilist
				}
				num_of_harmonized = 1
			}
			else = {
				religious_unity = 1
				prestige = 50
			}
		}
		effect = {
			if = {
				limit = { has_estate = estate_church }
				add_country_modifier = {
					name = itm_loyalists_control_the_church
					duration = 9125
				}
			}
			else = {
				add_adm_power = 100
			}
		}
	}
	itm_assuming_control = {
		icon = answer_the_roman_question
		required_missions = { itm_prelate_power }
		provinces_to_highlight = {
			province_id = 2235
			NOT = { owned_by = ROOT }
		}
		trigger = {
			OR = {
				is_papal_controller = yes
				is_elector = yes
				owns = 2235 # Rome
			}
		}
		effect = {
			add_country_modifier = {
				name = itm_italian_powerhouse
				duration = 10950
			}
		}
	}
}
