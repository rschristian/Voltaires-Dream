#Swedish LotN missions
VN_swedish_vanilla_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential_on_load = {
		has_dlc = "Lions of the North"
	}
	potential = {
		OR = {
			tag = SWE
			AND = {
				was_tag = SWE
				tag = SCA
			}
		}
		has_dlc = "Lions of the North"
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes

	swe_engelbrekt_rebellion = {
		icon = mission_engelbrekt
		position = 5
		required_missions = { }
		trigger = {
			if = {
				limit = { NOT = { started_in = 1444.11.11 } }
				is_year = 1410
				years_in_union_under = {
					who = DAN
					years = 30
				}
			}
			army_size_percentage = 0.75
			OR = {
				liberty_desire = 100
				any_hired_mercenary_company = {
					custom_trigger_tooltip = {
						tooltip = is_free_merc_company_tt
						OR = {
							template = merc__local_free_city_company
							template = merc__local_small_company
						}
					}
				}
			}
		}
		effect = {
			country_event = {
				id = flavor_swe.102
			}
		}
	}
	swe_swedish_allies = {
		icon = mission_alliances
		required_missions = { }
		provinces_to_highlight = {
			owner = {
				OR = {
					is_rival = DAN
					is_enemy = DAN
				}
			}
		}
		trigger = {
			if = {
				limit = { NOT = { started_in = 1444.11.11 } }
				is_year = 1410
				years_in_union_under = {
					who = DAN
					years = 30
				}
			}
			OR = {
				calc_true_if = {
					desc = swe_swedish_allies_at
					all_known_country = {
						OR = {
							is_rival = DAN
							is_enemy = DAN
						}
						has_opinion = {
							who = ROOT
							value = 100
						}
					}
					amount = 2
				}
				liberty_desire = 100
			}
		}
		effect = {
			country_event = {
				id = flavor_swe.106
			}
		}
	}
	swe_military_advisor = {
		icon = mission_swedish_military_advise
		required_missions = { }
		trigger = {
			if = {
				limit = { NOT = { started_in = 1444.11.11 } }
				is_year = 1410
				years_in_union_under = {
					who = DAN
					years = 30
				}
			}
			has_mil_advisor = yes
			OR = {
				custom_trigger_tooltip = {
					tooltip = swe_military_advisor_tt
					has_country_modifier = swe_vasa_leads_the_rebellion
				}
				liberty_desire = 100
				estate_territory = {
					estate = estate_nobles
					territory = 30
				}
			}
		}
		effect = {
			country_event = {
				id = flavor_swe.104
			}
		}
	}
	swe_gain_support_from_estates = {
		icon = mission_have_two_subjects
		required_missions = { }
		trigger = {
			if = {
				limit = { NOT = { started_in = 1444.11.11 } }
				is_year = 1410
				years_in_union_under = {
					who = DAN
					years = 30
				}
			}
			OR = {
				custom_trigger_tooltip = {
					tooltip = swe_gain_support_from_estates_ct
					OR = {
						AND = {
							num_of_estate_privileges = { estate = estate_nobles value = 3 }
							num_of_estate_privileges = { estate = estate_church value = 3 }
						}
						AND = {
							num_of_estate_privileges = { estate = estate_nobles value = 3 }
							num_of_estate_privileges = { estate = estate_burghers value = 3 }
						}
						AND = {
							num_of_estate_privileges = { estate = estate_church value = 3 }
							num_of_estate_privileges = { estate = estate_burghers value = 3 }
						}
					}
				}
				calc_true_if = {
					NOT = { has_estate = estate_nobles }
					NOT = { has_estate = estate_burghers }
					NOT = { has_estate = estate_church }
					amount = 2
				}
				AND = {
					is_subject = yes
					liberty_desire = 100
					custom_trigger_tooltip = {
						tooltip = swe_gain_support_from_estates_tt
						has_country_flag = dan_had_stockholm_bloodbath
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					has_country_flag = dan_had_stockholm_bloodbath
				}
				country_event = {
					id = flavor_swe.1080
				}
			}
			else = {
				custom_tooltip = swe_gain_support_from_estates_at
				country_event = {
					id = flavor_swe.108
				}
			}
		}
	}
}
