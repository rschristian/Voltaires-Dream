    vn_sar_develop_the_isles = {
		position = 1
		icon = develop_provence
		required_missions = { }
		provinces_to_highlight = {
			region = insular_region
			owned_by = ROOT
            OR = {
                NOT = { base_tax = 2 }
                NOT = { base_production = 2 }
                NOT = { base_manpower = 2 }
            }
        }
        trigger = {
			num_of_owned_provinces_with = {
				value = 10
				region = insular_region
				base_tax = 2
                base_production = 2
                base_manpower = 2
			}
        }
		effect = {
			add_legitimacy = 5
			capital_scope = {
				if = {
					limit = { has_port = yes }
					if = {
						limit = {
							NOT = { has_building = coastal_defence }
							NOT = { has_building = naval_battery }
						}
						add_building = coastal_defence
					}
					else_if = {
						limit = {
							has_building = coastal_defence
							NOT = { has_building = naval_battery }
						}
						remove_building = coastal_defence
						add_building = naval_battery
					}
					else = { add_base_manpower = 2 }
				}
				else = {
					if = {
						limit = {
							NOT = { has_building = customs_house }
							NOT = { has_building = arsenal }
						}
						add_building = customs_house
					}
					else_if = {
						limit = {
							has_building = customs_house
							NOT = { has_building = arsenal }
						}
						remove_building = customs_house
						add_building = arsenal
					}
					else = { add_base_manpower = 2 }
				}
			}
		}
    } 
    vn_sar_influential_estates = {
		icon = appease_the_estates
		required_missions = { vn_sar_develop_the_isles }
		trigger = {
            if = {
				limit = { has_estate = estate_nobles }
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
                estate_loyalty = {
                    estate = estate_nobles
                    loyalty = 60
                }
            }
            if = {
				limit = { has_estate = estate_burghers }
				estate_influence = {
					estate = estate_burghers
					influence = 60
				}
                estate_loyalty = {
                    estate = estate_burghers
                    loyalty = 60
                }
            }
            if = {
                limit = {
                    NOT = { has_estate = estate_nobles }
                    NOT = { has_estate = estate_burghers }
                }
                stability = 2
				if = {
					limit = { has_dlc = "Common Sense" }
					legitimacy_equivalent = 95
				}
				else_if = {
					limit = { government = monarchy }
					legitimacy = 95
				}
            }
		}
		effect = {
			add_adm_power = 50
			add_country_modifier = {
				name = sar_influential_estates_mission
				duration = 9125
			}
		}
    }
    vn_sar_gain_papal_support = {
        position = 4
		icon = sway_the_pope
		required_missions = { vn_sar_influential_estates }
		trigger = {
			if = {
				limit = { religion = catholic exists = PAP }
				OR = {
					AND = {
						reverse_has_opinion = {
							who = PAP
							value = 100
						}
						papal_influence = 50
					}
					AND = {
						power_projection = 50
						has_won_war_against = {
							who = PAP
							max_years_since = 25
						}
					}
				}
			}
			else_if = {
				limit = { religion = catholique exists = AVI }
				OR = {
					AND = {
						reverse_has_opinion = {
							who = AVI
							value = 100
						}
						papal_influence = 50
					}
					AND = {
						power_projection = 50
						has_won_war_against = {
							who = AVI
							max_years_since = 25
						}
					}
				}
			}
			else = {
				grown_by_development = 100
			}
		}
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							religion = catholic
							exists = PAP
						}
						AND = {
							religion = catholique
							exists = AVI
						}
					}
				}
				country_event = { id = vn_missions.14 }
			}
			else = {
				add_prestige_or_monarch_power = { amount = 10 } 
				add_legitimacy = 15
			}
		}
    }
}
