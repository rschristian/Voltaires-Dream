VN_hungarian_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		capital_scope = {
			superregion = carpathia_superregion
		}
		OR = {
			culture_group = magyar
			primary_culture = slovak
		}
		NOT = { tag = TEU }
		NOT = { map_setup = map_setup_random }
	}
	
	vn_hungarian_resettlement = {
		icon = mission_settlers_north_america
        required_missions = { }
        position = 5
		trigger = {
			capital_scope = {
				development = 20
			}
			num_of_owned_provinces_with = {
				value = 10
				development = 10
			}
		}
		effect = {
			add_prestige_or_monarch_power = { amount = 10 } 
			add_country_modifier = {
				name = "hungarian_resettlement_mission"
				duration = 9125
			}
		}
	}
	vn_legacy_of_the_magyars = {
		icon = mission_cossack_cavalry
		required_missions = { vn_hungarian_resettlement }
		trigger = {
			army_tradition = 50
			num_of_cavalry = 15
		}
		effect = {
			create_general = {
				tradition = 75
			}
			add_country_modifier = {
				name = "legacy_of_the_magyars_mission"
				duration = 9125
			}
		}
	}
	vn_danube_trade = {
		icon = mission_dominate_home_trade_node
		required_missions = { vn_legacy_of_the_magyars }
		trigger = {
			num_of_owned_provinces_with = {
				value = 10
				OR = {
					has_building = marketplace
					has_building = trade_depot
					has_building = stock_exchange
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "danube_trade_mission"
				duration = 9125
			}
		}
	}
	vn_universitiy_of_pecs = {
		icon = mission_village
		required_missions = { vn_danube_trade }
		provinces_to_highlight = {
			province_id = 2072
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { has_building = university }
			}
		}
		trigger = {
			if = {
				limit = { has_dlc = "Rule Britannia" }
				innovativeness = 20
			}
			2072 = {
				owned_by = ROOT
				has_building = university
			}
		}
		effect = {
			change_innovativeness = 1
			add_country_modifier = {
				name = "universitiy_of_pecs_mission"
				duration = 9125
			}
		}
	}
}
