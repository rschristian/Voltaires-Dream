#Hanseatic Cities
vn_gen_europe_3 = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		capital_scope = {
			continent = europe
			NOT = { superregion = caucasia_superregion }
			NOT = { superregion = anatolia_superregion }
		}
		is_part_of_hre = yes
		NOT = { has_reform = pirate_republic_reform }
		NOT = { primary_culture = eastphalian }
		NOT = { culture_group = latin }
	}
	has_country_shield = no
	
	vn_hanseatic_cities = {
		position = 11
		icon = mission_hanseatic_city
		required_missions = { }
		trigger = {
			num_of_owned_provinces_with = {
				value = 3
				has_province_modifier = hanseatic_city
			}
		}
		effect = {
			hidden_effect = {
				random_owned_province = {
					limit = {
						has_province_modifier = hanseatic_city
					}
					save_event_target_as = hanseatic_province
				}
			}
			if = {
				limit = { has_saved_event_target = hanseatic_province }
				event_target:hanseatic_province = {
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
					add_province_modifier = {
						name = "hanseatic_cities_mission"
						duration = 9125
					}
				}
			}
			else = {
				custom_tooltip = vn_hanseatic_cities.tt
			}
		}
	}
}

#HRE Monarchy
gen_europe_hre_monarchy = {
	slot = 2
	generic = yes
	ai = yes
	potential = {
		capital_scope = {
			continent = europe
			NOT = { superregion = caucasia_superregion }
			NOT = { superregion = anatolia_superregion }
		}
		is_part_of_hre = yes
		NOT = { government = theocracy }
		NOT = { government = republic }
	}
	has_country_shield = no
	
	gen_hre_electors = {
		icon = switzerland_and_the_empire
		required_missions = { }
		position = 9
		provinces_to_highlight = {
			is_capital = yes
			owner = {
				NOT = { tag = ROOT }
				is_elector = yes
			}
		}
		trigger = {
			if = {
				limit = {
					hre_size = 1
				}
				calc_true_if = {
					desc = gen_hre_electors.tooltip
					all_elector = {
						preferred_emperor = ROOT
					}
					amount = 2
				}
			}
			else = {
				num_of_allies = 3
			}
		}
		effect = {
			add_prestige_or_monarch_power = { amount = 10 }
			custom_tooltip = " " 
			add_country_modifier = {
				name = "gen_hre_diplomacy_mod"
				duration = 7300
			}
		}
	}
	gen_hre_emperor = {
		icon = holy_state
		required_missions = { gen_hre_electors }
		position = 10
		trigger = {
			if = {
				limit = {
					hre_size = 1
				}
				is_emperor = yes
			}
			else = {
				government_rank = 6
			}
		}
		effect = {
			add_country_modifier = {
				name = "gen_hre_emperor_mod"
				duration = 7300
			}
			define_advisor = {
				type = statesman
				skill = 2
				culture = ROOT
				religion = ROOT
				cost_multiplier = 0.5
			}
		}
	}
	gen_hre_imperial_ban = {
		icon = the_imperial_border
		required_missions = { gen_hre_emperor }
		position = 11
		provinces_to_highlight = {
			AND = {
				is_part_of_hre = yes
				owner = {
					is_part_of_hre = no
				}
			}
		}
		trigger = {
			if = {
				limit = {
					hre_size = 1
					NOT = { has_dlc = "Emperor" }
				}
				hre_reform_passed = reichsreform
				is_emperor = yes
				NOT = { num_of_foreign_hre_provinces = 1 }
			}
			else_if = {
				limit = {
					hre_size = 1
					has_dlc = "Emperor"
				}
				hre_reform_passed = emperor_reichsreform
				is_emperor = yes
				NOT = { num_of_foreign_hre_provinces = 1 }
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = imperial_ban_fallback_tt
					NOT = {
						any_core_province = {
							NOT = { owned_by = ROOT }
						}
					}
				}
			}
		}
		effect = {
			add_prestige_or_monarch_power = { amount = 10 } 
			if = {
				limit = {
					hre_size = 1
				}
				custom_tooltip = " "
				add_imperial_influence = 10
				add_country_modifier = {
					name = "gen_hre_imperial_ban_mod"
					duration = 7300
				}
			}
		}
	}
}

#HRE Theocracy
gen_europe_hre_theocracy = {
	slot = 2
	generic = yes
	ai = yes
	potential = {
		capital_scope = {
			continent = europe
			NOT = { superregion = caucasia_superregion }
			NOT = { superregion = anatolia_superregion }
		}
		is_part_of_hre = yes
		NOT = {
			is_emperor = yes
		}
		government = theocracy
	}
	has_country_shield = no
	
	gen_hre_theocracy_house_of_god = {
		icon = show_piety
		required_missions = { }
		position = 9
		provinces_to_highlight = {
			owned_by = root
			has_tax_building_trigger = no
		}
		trigger = {
			if = {
				limit = { NOT = { num_of_cities = 10 } }
				all_owned_province = {
					has_tax_building_trigger = yes
				}
			}
			else = {
				capital_scope = {
					has_tax_building_trigger = yes
				}
				num_of_owned_provinces_with = {
					value = 10
					has_tax_building_trigger = yes
				}
			}
		}
		effect = {
			capital_scope = {
				add_or_upgrade_tax_building = yes
			}
			if = {
				limit = {
					NOT = { legitimacy_equivalent = 85 }
				}
				add_devotion = 20
				add_legitimacy = 20
			}
			else = {
				add_stability_or_adm_power = yes
			}
		}
	}
	gen_hre_theocracy_lords_army = {
		icon = mission_knight
		required_missions = { }
		position = 10
		provinces_to_highlight = {
			owner = {
				is_neighbor_of = root
				army_size = root
			}
			root = {
				NOT = { army_size = 30 }
			}
		}
		trigger = {
			OR = {
				army_size = 30
				all_neighbor_country = {
					NOT = { army_size = root }
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = theocracy_lords_army_mod
				duration = 7300
			}
			if = {
				limit = {
					NOT = {
						army_tradition = 40
					}
				}
				create_general = { tradition = 40 }
			}
			else_if = {
				limit = {
					NOT = {
						army_tradition = 60
					}
				}
				create_general = { tradition = 60 }
			}
			else_if = {
				limit = {
					NOT = {
						army_tradition = 80
					}
				}
				create_general = { tradition = 80 }
			}
			else = {
				create_general = { tradition = 100 }
			}
		}
	}
	gen_hre_theocracy_divine_empire = {
		icon = counter_reformation
		required_missions = { gen_hre_theocracy_lords_army }
		position = 11
		provinces_to_highlight = {
			OR = {
				superregion = germanynorth_superregion
				superregion = germanysouth_superregion
			}
			NOT = { religion = root }
			NOT = {
				root = {
					calc_true_if = {
						amount = 250
						all_province = {
							OR = {
								superregion = germanynorth_superregion
								superregion = germanysouth_superregion
							}
							religion = root
						}
					}
				}
			}
		}
		trigger = {
			custom_trigger_tooltip = {
				tooltip = emp_bav_center_of_counter_reformation_tooltip2
				had_global_flag = {
					flag = reformation_started
					days = 18250
				}
			}
			calc_true_if = {
				desc = gen_hre_theocracy_divine_empire_trigger_tooltip
				amount = 250
				all_province = {
					OR = {
						superregion = germanynorth_superregion
						superregion = germanysouth_superregion
					}
					religion = root
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = theocracy_divine_empire_mod
				duration = 7300
			}
		}
	}
}
