vn_liv_missions_slot_2 = {
	slot = 7
	generic = no
	ai = yes
	potential = {
		tag = LIV
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	vn_liv_fight_for_christianity = {
		position = 1
		icon = chivalry_is_not_dead
		required_missions = { }
		trigger = {
			land_maintenance = 1
			army_size_percentage = 0.75
			religion_group = christian
			OR = {
				primary_culture = prussian
				primary_culture = baltic_german
			}
			OR = {
				is_year = 1350
				any_country = {
					war_with = ROOT
					NOT = { religion_group = christian }
				}
			}
		}
		effect = {
			add_army_tradition = 10
			if = {
				limit = {
					NOT = { tag = LIV }
					NOT = { is_rival = LIV }
					exists = LIV
				}
				LIV = {
					add_trust = {
						who = ROOT
						value = 10
					}
				}
			}
			else_if = {
				limit = {
					NOT = { tag = TEU }
					NOT = { is_rival = TEU }
					exists = TEU
				}
				TEU = {
					add_trust = {
						who = ROOT
						value = 10
					}
				}
			}
			latvian_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			livonia_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			estonia_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}

    vn_liv_convert_livonia = {
		position = 3
		icon = evengelize_the_faith
		required_missions = { vn_liv_conquer_latvia vn_liv_conquer_livonia vn_liv_conquer_estonia }
		provinces_to_highlight = {
			owned_by = ROOT
			religion_group = pagan
		}
		trigger = {
			religious_unity = 0.95
			NOT = {
				any_owned_province = {
					religion_group = pagan
				}
			}
		}
		effect = {
			add_prestige_or_monarch_power = { amount = 25 } 
			add_country_modifier = {
				name = teu_convert_prussia_mission
				duration = 18250
			}
		}
	}
	vn_liv_eternal_crusade = {
		icon = spread_piety
		required_missions = { vn_liv_convert_livonia vn_liv_conquer_curonia }
		trigger = {
			religion_group = christian
			OR = {
				full_idea_group = religious_ideas
				full_idea_group = theocracy_gov_ideas
			}
			religious_unity = 1
			if = {
				limit = { has_dlc = "Emperor" }
				OR = {
					is_defender_of_faith_of_tier = 5
					has_reform = religious_permanent_revolution_reform
				}
			}
			else = {
				OR = {
					is_defender_of_faith = yes
					has_reform = religious_permanent_revolution_reform
				}
			}
		}
		effect = {
			add_religious_influence_effect_large = yes
			if = {
				limit = {
					has_dlc = "Cradle of Civilization"
					OR = {
						religion = catholic
						religion = catholique
					}
				}
				set_country_flag = can_use_propagate_religion
				custom_tooltip = vn_teu_eternal_crusade.tt
			}
		}
	}
}
