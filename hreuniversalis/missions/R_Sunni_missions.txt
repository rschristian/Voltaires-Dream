vn_sunni_missions = {
	slot = 3
	generic = yes
	ai = yes
	potential = {
		religion = sunni
	}
	
	vn_expand_our_harem = {
		position = 9
		icon = mission_ottoman_harem
		required_missions = { }
		trigger = {
			if = {
				limit = { government = monarchy }
				num_of_royal_marriages = 3
			}
			else = {
				num_of_allies = 3
			}
			prestige = 40
		}
		effect = {
			add_country_modifier = {
				name = "expand_our_harem_mission"
				duration = 9125
			}
		}
	}
	vn_legalism_or_mysticism = {
		icon = mission_mosque
		required_missions = { vn_expand_our_harem }
		trigger = {
			if = {
				limit = {
					uses_piety = yes
				}
				OR = {
					NOT = { piety = -0.9 }
					piety = 0.9
				}
			}
			else = {
				religious_unity = 1
			}
			prestige = 60
		}
		effect = {
			add_country_modifier = {
				name = "legalism_or_mysticism_mission"
				duration = 9125
			}
		}
	}
	vn_fight_a_jihad = {
		icon = mission_persian_soldiers
		required_missions = { }
		trigger = {
			custom_trigger_tooltip = {
				tooltip = vn_fight_a_jihad.tt
				any_country = {
					NOT = { religion_group = ROOT }
					is_in_war = {
						attacker_leader = ROOT
					}
					ROOT = {
		                war_score_against = {
		                    who = PREV
		                    value = 75
		                }
		            }
				}
			}
		}
		effect = {
			add_religious_influence_effect_large = yes
			add_country_modifier = {
				name = "fight_a_jihad_mission"
				duration = 9125
			}
		}
	}
}
