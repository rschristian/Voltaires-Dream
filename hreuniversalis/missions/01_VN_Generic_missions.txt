		}
		trigger = {
			if = {
				limit = {
					started_in = 1250.1.1

				limit = {
					started_in = 1250.1.1

					value = 10
					has_building = post_office
				}

					has_institution = guilds

					}

			}
			else = {
				OR = {
					has_institution = guilds

						value = 10

				}
			}
		}
		effect = {
			add_years_of_owned_provinces_production_income = {
				years = 1.5

					always = yes

			custom_tooltip = " "
			add_country_modifier = {
				name = "embrace_guilds_mission"
				duration = 9125
			}

							owned_by = ROOT

						save_event_target_as = reward_province

						limit = {
							owned_by = ROOT

						limit = {
							owned_by = ROOT
							development = CAPITAL
							is_capital = no
							development = 30
						}
						save_event_target_as = reward_province
					}
				}
			}
			if = {
				limit = { has_saved_event_target = reward_province }

					add_province_modifier = {
						name = "dominant_trade_hub"
						duration = -1
					}
				}
			}
			else = {
				custom_tooltip = dominant_home_node_reward_tooltip
			}
			custom_tooltip = " "
			add_mercantilism_or_monarch_power = { amount = 3 }
		}
	}
