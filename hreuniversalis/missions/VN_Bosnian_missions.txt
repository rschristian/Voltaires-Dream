    vn_bos_dinaric_trade_routes = {
        #icon = 
        position = 1
        required_missions = {  }
        provinces_to_highlight = {
            province_id = 890
            OR = {
                NOT = { owned_by = ROOT }
                NOT = { is_core = ROOT }
            }
        }
        trigger = {
            owns_core_province = 890
            539 = {
				trade_share = {

				}

        effect = {
            890 = {
                add_or_upgrade_trade_building = yes
                add_province_modifier = {
                    name = bos_dinaric_trade_routes_mission
                    duration = -1
                }
            }
        }
    }
    vn_bos_enforce_bogomilism = {
        #icon = 
        position = 2
        required_missions = {  }
        trigger = {
            religion = bogomilist

				limit = { 
					has_dlc = "Mandate of Heaven" 
					religion = bogomilist 
				} 
				harmony = 90 
			}
            else = {
				religious_unity = 0.9
			}
        }
        effect = {
            
        }
    }
    vn_bos_move_against_dubrovnik = {
        #icon = 
        position = 3
        required_missions = { vn_bos_enforce_bogomilism }
        provinces_to_highlight = {
            province_id = 137
            NOT = { country_or_non_sovereign_subject_holds = ROOT }
        }
        trigger = {
            137 = { country_or_non_sovereign_subject_holds = ROOT }
        }
        effect = {
            137 = {
                add_unit_construction = {
					type = light_ship
					amount = 5
					speed = 0.1
					cost = 0
				}
            }
        }
    }
    vn_bos_bastion_of_bogomilism = {
        icon = spread_piety
        position = 4
        required_missions = { vn_bos_move_against_dubrovnik }
        trigger = {
            
        }
        effect = {
            if = {
				limit = {
					exists = SER
					valid_for_personal_unions_trigger = yes
					SER = {
						valid_for_personal_unions_trigger = yes
					}
				}
				add_casus_belli = {
					type = cb_fabricated_claims
					months = 300
					target = SER
				}
			}
            else_if = {
                limit = {
					exists = KSV
					valid_for_personal_unions_trigger = yes
					KSV = {
						valid_for_personal_unions_trigger = yes
					}
				}
				add_casus_belli = {
					type = cb_fabricated_claims
					months = 300
					target = KSV
				}
            }
            else = {
                serbia_region = {
                    limit = {
                        NOT = { is_core = ROOT }
                        NOT = { is_permanent_claim = ROOT }
                    }
                    add_permanent_claim = ROOT
                }
            }
        }
    }
    vn_bos_serbian_throne = {
        icon = defeat_the_serbs
        position = 5
        required_missions = { vn_bos_bastion_of_bogomilism }
        provinces_to_highlight = {
            region = serbia_region
            NOT = { country_or_non_sovereign_subject_holds = ROOT }
        }
        trigger = {
            serbia_region = {
                type = all
				country_or_non_sovereign_subject_holds = ROOT
            }
        }
        effect = {
            add_power_projection = {
				type = mission_rewards_power_projection
				amount = 50
			}
            if = {
				limit = {
					SER = {
						is_subject_of_type = personal_union
						is_subject_of = ROOT
						NOT = { religion = ROOT }
					}
				}
				SER = {
					change_religion = ROOT
				}
			}
			else_if = {
				limit = {
					KSV = {
						is_subject_of_type = personal_union
						is_subject_of = ROOT
						NOT = { religion = ROOT }
					}
				}
				KSV = {
					change_religion = ROOT
				}
			}
        }
    }
    vn_bos_tvrtkos_claim = {
