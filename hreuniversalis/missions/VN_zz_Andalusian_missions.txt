                            tooltip = has_been_insulted_by_root

                                has_opinion_modifier = {
                                    who = ROOT
                                    modifier = insulted
                                }
                                has_opinion_modifier = {
                                    who = ROOT
                                    modifier = scornfully_insulted
                                }
                            }

                            has_won_war_against = {

    vn_taifa_spoils_of_war = {
        position = 4
