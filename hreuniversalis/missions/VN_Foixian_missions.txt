        icon = mission_rb_scottish_conquer_london
        required_missions = { }
        trigger = {
	    	dip = 4
	    	diplomatic_reputation = 4
        }
        effect = {
            add_dip_power = 50
			if = {
				limit = { NOT = { dip = 6 } }
				change_dip = 1
			}
			else_if = {
				limit = { has_heir = yes NOT = { heir_dip = 6 } }
				change_heir_dip = 1
			}
			else = {
				add_prestige = 20
			}
        }
    }
    vn_FOI_crucial_french_ally = {
        position = 3
        icon = mission_buildings
        required_missions = { vn_FOI_good_diplomats }
        trigger = {
            if = {
                limit = {
                    is_subject_of = FRA
                }
                FRA = {
                    has_opinion = {
                        who = ROOT
                        value = 125
                    }
                }
                OR = {
                    is_in_war = {
                        defender_leader = FRA
                        war_score = 75
                    }
                    is_in_war = {
                        attacker_leader = FRA
                        war_score = 75
                    }
                }
            }
            else = {
                army_size = 25
                manpower_percentage = 0.75
            }
        }
        effect = {
			add_dip_power = 100
            add_country_modifier = {
                name = FOI_crucial_french_ally_mission
                duration = 9125
            }
        }
    }
    vn_FOI_powerful_prince = {
        position = 4
        icon = mission_buildings
        required_missions = { vn_FOI_crucial_french_ally }
        trigger = {
            if = {
                limit = { has_dlc = "Dharma" }
                has_reform_level = 4
            }
            OR = {
                NOT = { exists = FRA }
                army_strength = {
                    who = FRA
                    value = 0.75
                }
            }
        }
        effect = {
            add_army_tradition = 25

                limit = {
                    valid_for_personal_unions_trigger = yes

						valid_for_personal_unions_trigger = yes
					}
                }
                add_casus_belli = {
                    type = cb_fabricated_claims

                    target = NEM
                }
            }
            else = {
                nemours_area = {
                    limit = {
                        NOT = { is_core = ROOT }
                    }
                    add_core = ROOT
                }
            }
        }
    }
    vn_FOI_vassals_in_name = {
