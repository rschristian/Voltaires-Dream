VN_breton_minors_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		primary_culture = breton
		capital_scope = {
			region = brittany_region
		}
		NOT = { map_setup = map_setup_random }
	}
	
	vn_strengthen_agriculture = {
		icon = mission_have_manufactories
        required_missions = { }
        position = 5
		provinces_to_highlight = {
			owned_by = ROOT
			trade_goods = livestock
			NOT = { development = 10 }
		}
		trigger = {
			num_of_owned_provinces_with = {
				value = 5
				trade_goods = livestock
				development = 10
			}
		}
		effect = {
			add_country_modifier = {
				name = "strengthen_agriculture_mission"
				duration = 9125
			}
		}
	}
	vn_breton_fishery = {
		icon = mission_trade_company_region_abroad
		required_missions = { vn_strengthen_agriculture }
		trigger = {
			naval_forcelimit = 12
			num_of_light_ship = 10
		}
		effect = {
			add_country_modifier = {
				name = "breton_fishery_mission"
				duration = 9125
			}
		}
	}
	vn_breton_alliances = {
		icon = mission_alliances
		required_missions = { vn_breton_fishery }
		trigger = {
			calc_true_if = {
				all_ally = {
					total_development = 150
				}
				amount = 3
			}
		}
		effect = {
			add_country_modifier = {
				name = "breton_alliances_mission"
				duration = 9125
			}
		}
	}
	vn_military_port_of_brest = {
		icon = mission_sea_battles
		required_missions = { vn_breton_alliances }
		provinces_to_highlight = {
			province_id = 2294
		}
		trigger = {
			2294 = {
				owned_by = ROOT
				development = 20
				has_building = canal
				OR = {
					has_building = dock
					has_building = drydock
				}
				OR = {
					has_building = shipyard
					has_building = grand_shipyard
				}
			}
		}
		effect = {
			2294 = {
				if = {
					limit = {
						province_has_center_of_trade_of_level = 1
						NOT = { province_has_center_of_trade_of_level = 3 }
					}
					add_center_of_trade_level = 1
				}
				else_if = {
					limit = {
						NOT = { province_has_center_of_trade_of_level = 1 }
					}
					center_of_trade = 1
				}
				else = {
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
				}
			}
			add_country_modifier = {
				name = "military_port_of_brest_mission"
				duration = 9125
			}
		}	
	}
}

VN_breton_conquest_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		always = no
		primary_culture = breton
		capital_scope = {
			region = brittany_region
		}
		NOT = { map_setup = map_setup_random }
	}

	vn_breton_unity = {
		position = 9
		icon = mission_landsknecht_soldier
		required_missions = {}
		provinces_to_highlight = {
			region = brittany_region
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
			}
		}
		trigger = {
			brittany_region = {
				type = all
				owned_by = ROOT
				is_core = ROOT
			}
		}
		effect = {
			create_general = {
				tradition = 75
				culture = breton
			}
			add_country_modifier = {
				name = "breton_unity_mission"
				duration = 9125
			}
		}
	}
	vn_breton_normandy = {
		icon = mission_soldiers_in_line
		required_missions = { vn_breton_unity }
		provinces_to_highlight = {
			region = normandy_region
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		trigger = {
			normandy_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			normandy_region = {
				limit = { country_or_non_sovereign_subject_holds = ROOT }
				add_province_modifier = {
					name = "breton_normandy_mission"
					duration = 7300
				}
			}
			if = {
				limit = { NOT = { owns = 320 } }
				320 = { add_permanent_claim = ROOT }
			}
			if = {
				limit = { NOT = { owns = 2141 } }
				2141 = { add_permanent_claim = ROOT }
			}
		}
	}
	vn_liberate_cornwall = {
		icon = mission_unite_home_region
		required_missions = { vn_breton_normandy }
		provinces_to_highlight = {
			OR = {
				province_id = 320
				province_id = 2141
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		trigger = {
			320 = { country_or_non_sovereign_subject_holds = ROOT }
			2141 = { country_or_non_sovereign_subject_holds = ROOT }
		}
		effect = {
			add_prestige = 10
			add_country_modifier = {
				name = "liberate_cornwall_mission"
				duration = 9125
			}
		}
	}
}
