vn_pfu_missions_slot_1 = {
    slot = 6
    generic = no
    ai = yes
    potential = {
        tag = PFU
        NOT = { map_setup = map_setup_random }
    }
    has_country_shield = yes

    vn_pfu_swabian_league = {
        icon = mission_hanseatic_city
        position = 3
        required_missions = { vn_pfu_adopt_a_constitution }
        provinces_to_highlight = {
            is_capital_of = emperor
        }
        trigger = {
			if = {
				limit = { hre_size = 1 }
				OR = {
					AND = {
						calc_true_if = {
							amount = 5
							all_ally = {
								capital_scope = {
									OR = {
										superregion = germanynorth_superregion
										superregion = germanysouth_superregion
									}
								}
							}
						}
						war_with = emperor
					}
					is_emperor = yes
				}
			}
			army_size = 20
        }
        effect = {
			add_army_tradition = 10
			add_power_projection = {
				type = mission_rewards_power_projection
				amount = 10
			}
            add_country_modifier = {
				name = pfu_swabian_league_mission
				duration = 9125
			}
        }
	}
}

vn_pfu_missions_slot_2 = {
    slot = 7
    generic = no
    ai = yes
    potential = {
        tag = PFU
        NOT = { map_setup = map_setup_random }
    }
    has_country_shield = yes

    vn_pfu_blutgerichtsbarkeit = {
        icon = switzerland_and_the_empire
        position = 1
        required_missions = {  }
        provinces_to_highlight = {
            is_capital_of = emperor
        }
        trigger = {
            OR = {
				reverse_has_opinion = {
					who = emperor
					value = 125
				}
				is_emperor = yes
				NOT = { hre_size = 1 }
			}
        }
        effect = {
			custom_tooltip = vn_pfu_blutgerichtsbarkeit.tt
			hidden_effect = {
				if = {
					limit = {
						alliance_with = emperor
					}
					emperor = {
						add_trust = {
							who = ROOT
							value = 10
						}
					}
				}
			}
            add_country_modifier = {
				name = pfu_blutgerichtsbarkeit_mission
				duration = 9125
			}
        }
	}
	vn_pfu_adopt_a_constitution = {
        icon = republican_yearnings
        position = 2
		required_missions = { vn_pfu_blutgerichtsbarkeit }
		provinces_to_highlight = {
			province_id = 1875
			NOT = { owned_by = ROOT }
		}
        trigger = {
			owns = 1875
            OR = {
				AND = {
					government = republic
					republican_tradition = 100
				}
				AND = {
					NOT = { government = republic }
					has_estate = estate_burghers
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 60
					}
					estate_influence = {
						estate = estate_burghers
						influence = 60
					}
					num_of_estate_agendas_completed = {
						estate = estate_burghers
						value = 5
					}
				}
			}
        }
        effect = {
            if = {
				limit = {
					1875 = {
						NOT = { province_has_center_of_trade_of_level = 1 }
					}
				}
				1875 = {
					center_of_trade = 1
				}
			}
			else_if = {
				limit = {
					1875 = {
						NOT = { province_has_center_of_trade_of_level = 3 }
					}
				}
				1875 = {
					add_center_of_trade_level = 1
				}
			}
			else = {
				1875 = { add_base_production = 2 }
			}
			1875 = {
				add_province_modifier = {
					name = pfu_adopt_a_constitution_mission
					duration = -1
				}
			}
        }
	}
	vn_pfu_herrschaft_der_gilden = {
        icon = mission_hanseatic_city
        position = 4
        required_missions = { vn_pfu_adopt_a_constitution }
        provinces_to_highlight = {
            province_id = 1875
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { development = 30 }
			}
        }
        trigger = {
			1875 = {
				owned_by = ROOT
				development = 30
				if = {
					limit = { has_dlc = "Dharma" }
					province_has_center_of_trade_of_level = 3
				}
				else = {
					NOT = { num_free_building_slots = 1 }
				}
			}
            home_trade_node = {
				is_strongest_trade_power = ROOT
			}
			num_of_owned_provinces_with = {
				value = 2
				province_has_center_of_trade_of_level = 3
			}
        }
        effect = {
			add_mercantilism = 2
            add_country_modifier = {
				name = pfu_herrschaft_der_gilden_mission
				duration = 9125
			}
        }
	}
}
