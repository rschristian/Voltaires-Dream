#VN Mediatization Incident

namespace = incident_mediatization

#Initial Event
country_event = {
	id = incident_mediatization.1
	title = incident_mediatization.1.t
	desc = incident_mediatization.1.d
	picture = HRE_eventPicture
	
	fire_only_once = yes 
	major = yes
	
	trigger = {
		has_dlc = "Emperor"
		is_institution_enabled = enlightenment
		num_of_foreign_hre_provinces = 100
		NOT = { started_in = 1789.7.14 }
		NOT = { active_imperial_incident = any }
		is_part_of_hre = no
		is_subject = no
		NOT = { has_country_flag = med_demander_country }
		any_country = {
			is_part_of_hre = yes
			government = theocracy
		}
		any_country = {
			is_part_of_hre = yes
			has_reform = free_city
		}
		OR = {
			any_owned_province = {
				is_part_of_hre = yes
			}
			any_subject_country = {
				any_owned_province = {
					is_part_of_hre = yes
				}
			}
		}
		NOT = {
			any_known_country = {
				is_part_of_hre = no
				is_subject = no
				OR = {
					any_owned_province = {
						is_part_of_hre = yes
					}
					any_subject_country = {
						any_owned_province = {
							is_part_of_hre = yes
						}
					}
				}
				num_of_cities = ROOT
			}
		}
	}
	
	mean_time_to_happen = {
		months = 300
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = med_demander_country
			save_global_event_target_as = med_demander
		}
	}
	
	option = {
		name = incident_mediatization.1.a
		ai_chance = { factor = 100 }
		custom_tooltip = incident_mediatization.1.a.tt
		hidden_effect = {
			every_country = {
				limit = {
					is_part_of_hre = yes
					NOT = { has_reform = papacy_reform }
					OR = {
						government = theocracy
						has_reform = free_city
					}
				}
				country_event = {
					id = incident_mediatization.2
					days = 1
				}
			}
		}
		set_imperial_incident = incident_mediatization
	}
	
	option = {
		name = incident_mediatization.1.b
		ai_chance = { factor = 0 }
		add_prestige = -20
	}
}
