		}
	}
}

#Spawn rebels
country_event = {
	id = seljuk_collapse.2
	title = seljuk_collapse.2.t
	desc = seljuk_collapse.2.d
	picture = REVOLT_eventPicture

	goto = sjk_rebels_prov
	is_triggered_only = yes
	
	trigger = {
		any_owned_province = {
			can_spawn_rebels = nationalist_rebels
			controlled_by = ROOT
			any_core_country = {
				NOT = { tag = ROOT }
				exists = no
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					can_spawn_rebels = nationalist_rebels
					controlled_by = ROOT
					any_core_country = {
						NOT = { tag = ROOT }
						exists = no
					}
				}
				save_event_target_as = sjk_rebels_prov
			}
		}
	}
	
	option = {
		name = seljuk_collapse.2.a
		event_target:sjk_rebels_prov = {
			spawn_rebels = {
				type = nationalist_rebels 
				size = 3
			}
		}
	}
}

# Yazd declares independence
country_event = {
	id = seljuk_collapse.3
	title = seljuk_collapse.3.t
	desc = seljuk_collapse.3.d
	picture = CIVIL_WAR_eventPicture
	goto = YZD_capital
	
	fire_only_once = yes
	is_triggered_only = yes 
	
	trigger = {
		NOT = { exists = YZD }
		any_owned_province = {
			area = yazd_area
		}
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		hidden_effect = {
			yazd_area = {
				limit = {
					owned_by = ROOT
				}
				add_core = KHO
			}
			release = YZD
			YZD = {
				change_government = monarchy
				add_government_reform = iqta
				if = {
					limit = { is_part_of_hre = yes }
					set_in_empire = no
				}
				set_government_rank = 3
				add_stability = 3
				change_religion = sunni
				set_religious_school = {
					group = muslim
					school = jafari_school
				}
				define_ruler = {
					name = "Langar"
					dynasty = "Wardanruzi"
					adm = 4
					dip = 3
					mil = 3
				}
				remove_heir = {}
				random_owned_province = {
					build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
				capital_scope = {
					save_event_target_as = YZD_capital
				}
			}
		}
	}

	option = {
		name = "seljuk_collapse.3.a"
		ai_chance = { factor = 1 }
		custom_tooltip = seljuk_collapse.tt
		add_truce_with = YZD
		hidden_effect = {
			YZD = {
				every_owned_province = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
		}
	}
	
	option = {
		name = "seljuk_collapse.3.b"
		ai_chance = { factor = 0 }
		declare_war_with_cb = {
			who = YZD
			casus_belli = cb_disloyal_vassal
		}
	}

	option = {
		name = "seljuk_collapse.3.c"
		trigger = { ai = no }
		ai_chance = { factor = 0 }
		switch_tag = YZD
		add_truce_with = YZD
		hidden_effect = {
			YZD = {
				every_owned_province = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
		}
	}
}
